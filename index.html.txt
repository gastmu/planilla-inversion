<script>
function calcular() {
    const filas = document.querySelectorAll("#tabla tbody tr");
    const mesFiltro = document.getElementById("filtroMes").value;

    let saldo = 0;
    let saldoPeriodo = 0;
    let capInicial = null;
    let aportes = 0;
    let aportesTotales = 0;
    let movPeriodo = 0;

    filas.forEach(fila => {
        const mes = fila.children[0].children[0].value;
        const aporte = Number(fila.children[3].children[0].value) || 0;
        const retiro = Number(fila.children[4].children[0].value) || 0;

        saldo += aporte - retiro;
        aportesTotales += aporte;

        if (!mesFiltro || mesFiltro === mes) {
            if (capInicial === null) capInicial = saldo - (aporte - retiro);
            saldoPeriodo += aporte - retiro;
            aportes += aporte;
            movPeriodo++;
        }

        fila.querySelector(".saldo").innerText = "$" + saldo.toFixed(2);
    });

    // Resumen general
    document.getElementById("capInvertido").innerText = "$" + aportesTotales.toFixed(2);
    document.getElementById("capActual").innerText = "$" + saldo.toFixed(2);
    document.getElementById("ganancia").innerText = "$" + (saldo - aportesTotales).toFixed(2);
    document.getElementById("rentTotal").innerText =
        aportesTotales > 0 ? ((saldo - aportesTotales) / aportesTotales * 100).toFixed(2) + "%" : "0%";
    document.getElementById("movs").innerText = filas.length;

    // Resumen por período
    const resultado = saldoPeriodo;
    const rentPeriodo = aportes > 0 ? (resultado / aportes * 100) : 0;

    document.getElementById("capInicial").innerText = "$" + (capInicial ?? 0).toFixed(2);
    document.getElementById("aportes").innerText = "$" + aportes.toFixed(2);
    document.getElementById("resultado").innerText = "$" + resultado.toFixed(2);
    document.getElementById("rentPeriodo").innerText = rentPeriodo.toFixed(2) + "%";
    document.getElementById("movPeriodo").innerText = movPeriodo;
}
</script>
